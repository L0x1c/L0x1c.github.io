function print_array_by_reflect(ptr) {
    if (ptr.isNull()) return;
    Java.perform(function () {
        try {
            var obj = Java.cast(ptr, Java.use("java.lang.Object"));
            if (!obj.getClass().isArray()) {
                console.log("Not an array: " + obj.toString());
                return;
            }
            var ArrayClass = Java.use("java.lang.reflect.Array");
            var length = ArrayClass.getLength(obj);

            results = [];
            for (var i = 0; i < length; i++) {
                var item = ArrayClass.get(obj, i);
                if (item === null) {
                    results.push("null");
                } else {
                    results.push(item.toString());
                }
            }

            console.log("\n[+] Array Dump (" + length + "): " + JSON.stringify(results));

        } catch (e) {
            console.error("Reflect error: " + e);
        }
    });
}

#!/usr/bin/env python3
"""
从 ARM64 指令 trace 日志中提取 SHA1 算法的输入和输出。

原理:
  1. 搜索 SHA1 轮常数 K[0]=0x5a827999 的 add 指令，提取每轮的 x14 值
  2. 搜索 SHA1 初始化常数 0x67452301 的加载，确定 SHA1 调用边界
  3. 根据 SHA1 轮函数反推每个 block 的 16 个消息字 W[0]-W[15]
  4. 从最后一个 block 的 padding 字段提取消息长度
  5. 从最后一个 block 结束后的 add 指令提取最终 hash

用法:
  python3 extract_sha1.py trace_2.asm
"""

import re
import sys
import hashlib
from pathlib import Path

# ──────────────────── SHA1 辅助函数 ────────────────────

MASK32 = 0xFFFFFFFF

SHA1_K = [0x5A827999, 0x6ED9EBA1, 0x8F1BBCDC, 0xCA62C1D6]
SHA1_H0 = [0x67452301, 0xEFCDAB89, 0x98BADCFE, 0x10325476, 0xC3D2E1F0]


def rotl32(x, n):
    return ((x << n) | (x >> (32 - n))) & MASK32


def sha1_f(t, b, c, d):
    if t < 20:
        return ((b & c) | (~b & d)) & MASK32
    elif t < 40:
        return (b ^ c ^ d) & MASK32
    elif t < 60:
        return ((b & c) | (b & d) | (c & d)) & MASK32
    else:
        return (b ^ c ^ d) & MASK32


def sha1_K(t):
    return SHA1_K[t // 20]


# ──────────────────── Trace 解析 ────────────────────

# 匹配: add x16, x14, x15  ... X15=0x5a827999  X14=0x??? ... | X16=0x???
RE_ADD_K0 = re.compile(
    r"add\s+x16,\s*x14,\s*x15\s+"
    r".*X15=0x5a827999\s+"
    r"X14=(0x[0-9a-fA-F]+)\s+"
    r".*\|\s+X16=(0x[0-9a-fA-F]+)"
)

# 匹配所有四个 K 常数的 add 指令
RE_ADD_K_ALL = re.compile(
    r"add\s+x16,\s*x14,\s*x15\s+"
    r".*X15=(0x(?:5a827999|6ed9eba1|(?:ffffffff)?8f1bbcdc|(?:ffffffff)?ca62c1d6))\s+"
    r"X14=(0x[0-9a-fA-F]+)\s+"
    r".*\|\s+X16=(0x[0-9a-fA-F]+)",
    re.IGNORECASE,
)

# 匹配 SHA1 初始化: ldr w0, ... | X0=0x67452301
RE_SHA1_INIT = re.compile(
    r"ldr\s+w\d+,.*\|\s+\w+=0x67452301\b"
)

# 匹配 H[4] 初始值用于最终 add: X15=0xc3d2e1f0
RE_ADD_H4 = re.compile(
    r"add\s+x16,\s*x14,\s*x15\s+"
    r".*X15=0xc3d2e1f0\s+"
    r"X14=(0x[0-9a-fA-F]+)\s+"
    r".*\|\s+X16=(0x[0-9a-fA-F]+)"
)


def scan_trace(filepath):
    """扫描 trace 文件，提取 SHA1 相关的关键行。"""
    print(f"[*] 扫描 trace 文件: {filepath}")

    k0_adds = []       # (line_no, x14, x16) for K[0] rounds
    k_all_adds = []    # (line_no, K_val, x14, x16) for all K rounds
    init_lines = []    # line numbers where 0x67452301 is loaded

    with open(filepath, "r", errors="replace") as f:
        for line_no, line in enumerate(f, 1):
            if line_no % 1_000_000 == 0:
                print(f"    ... 已扫描 {line_no // 1_000_000}M 行")

            # K[0] add 指令 (用于提取 W 值)
            m = RE_ADD_K0.search(line)
            if m:
                x14 = int(m.group(1), 16)
                x16 = int(m.group(2), 16)
                k0_adds.append((line_no, x14, x16))
                continue

            # 所有 K 常数的 add (用于定界 block)
            m = RE_ADD_K_ALL.search(line)
            if m:
                k_val = int(m.group(1), 16) & MASK32
                x14 = int(m.group(2), 16)
                x16 = int(m.group(3), 16)
                k_all_adds.append((line_no, k_val, x14, x16))
                continue

            # SHA1 init
            if "0x67452301" in line and RE_SHA1_INIT.search(line):
                init_lines.append(line_no)

    print(f"[*] 扫描完成:")
    print(f"    SHA1 初始化次数: {len(init_lines)}")
    print(f"    K[0] add 指令数: {len(k0_adds)}")
    print(f"    所有 K add 指令数: {len(k_all_adds)}")

    return k0_adds, k_all_adds, init_lines


def group_into_blocks(k0_adds):
    """将 K[0] 的 20 轮 add 按 block 分组。

    每个 SHA1 block 有 20 轮使用 K[0]。
    通过行号间隔判断 block 边界。
    """
    if not k0_adds:
        return []

    blocks = []
    current_block = [k0_adds[0]]

    for i in range(1, len(k0_adds)):
        prev_line = k0_adds[i - 1][0]
        curr_line = k0_adds[i][0]
        # 同一 block 内轮间距约 250 行，block 间距 > 20000 行
        if curr_line - prev_line > 10000:
            blocks.append(current_block)
            current_block = [k0_adds[i]]
        else:
            current_block.append(k0_adds[i])

    blocks.append(current_block)
    return blocks


def extract_W_from_block(x14_values, state):
    """根据 K[0] 轮的 x14 值和初始 state，反推 W[0]-W[15]。

    在 SHA1 第 t 轮:
      temp = ROTL5(a) + f(t,b,c,d) + e + W[t] + K[t]

    VM 执行: x14 = ROTL5(a) + f(t,b,c,d) + e + W[t]  (64 位累加)
             x16 = x14 + K[t]

    因此: W[t] = (x14 - (ROTL5(a) + f(t,b,c,d) + e)) & 0xFFFFFFFF
    """
    a, b, c, d, e = state
    W = []

    for t in range(min(16, len(x14_values))):
        rotl5_a = rotl32(a, 5)
        f = sha1_f(t, b, c, d)
        partial_sum = rotl5_a + f + e  # 64 位累加

        w = (x14_values[t] - partial_sum) & MASK32
        W.append(w)

        # 更新 state 进入下一轮
        temp = (x14_values[t] + SHA1_K[0]) & MASK32
        a, b, c, d, e = temp, a, rotl32(b, 30), c, d

    return W

    """

    # 结果 x16 的低 32 位就是新的 H[i]

def find_final_hash_adds(filepath, last_k_line):
    """在最后一个 K 常数 add 之后，查找 H[0]-H[4] 的最终累加。

    特征: add x16, x14, x15 其中 x15 是上一个 block 的 hash state 值。
    最后的 add 序列会把 c3d2e1f0 (H[4] 初始值) 或 block 中间 hash 加回。
    """
    print(f"[*] 在行 {last_k_line} 之后搜索最终 hash 累加...")

    # 搜索模式: 5 个连续 add 指令，间隔较近
    re_add = re.compile(
        r"add\s+x16,\s*x14,\s*x15\s+"
        r".*X15=(0x[0-9a-fA-F]+)\s+"
        r"X14=(0x[0-9a-fA-F]+)\s+"
        r".*\|\s+X16=(0x[0-9a-fA-F]+)"
    )

    candidates = []
    with open(filepath, "r", errors="replace") as f:
        for line_no, line in enumerate(f, 1):
            if line_no <= last_k_line:
                continue
            if line_no > last_k_line + 5000:
                break
            m = re_add.search(line)
            if m:
                x15 = int(m.group(1), 16)
                x14 = int(m.group(2), 16)
                x16 = int(m.group(3), 16)
                candidates.append((line_no, x15, x14, x16))

    return candidates


def find_block_final_adds(filepath, block_k_all_adds):
    """查找每个 block 结束后的 5 个 hash 累加 add 指令。

    block 结束后的 add 序列特征:
    - x15 = 当前 block 的 H[i] (对第一个 block 是 SHA1 初始值)
    - x14 = 工作变量
    - x16 = H[i]_new = x14 + x15
    """
    if not block_k_all_adds:
        return []

    # 找到最后一个 K 常数 add 的行号
    last_line = block_k_all_adds[-1][0]
    candidates = find_final_hash_adds(filepath, last_line)

    # 过滤: 查找连续的 5 个 add，其中 x15 是已知的 H 值
    return candidates


# ──────────────────── 主逻辑 ────────────────────

def main():
    if len(sys.argv) < 2:
        print(f"用法: {sys.argv[0]} <trace_file>")
        sys.exit(1)

    filepath = sys.argv[1]
    if not Path(filepath).exists():
        print(f"错误: 文件不存在: {filepath}")
        sys.exit(1)

    # ───── 第一步: 扫描 trace ─────
    k0_adds, k_all_adds, init_lines = scan_trace(filepath)

    # ───── 第二步: 将 K[0] 轮按 block 分组 ─────
    blocks = group_into_blocks(k0_adds)
    num_blocks = len(blocks)
    print(f"\n[*] 检测到 {num_blocks} 个 SHA1 block")
    for i, blk in enumerate(blocks):
        print(f"    Block {i}: {len(blk)} 轮 K[0] add, "
              f"行号 {blk[0][0]}-{blk[-1][0]}")

    if num_blocks == 0:
        print("[!] 未检测到 SHA1 计算")
        sys.exit(1)

    # ───── 第三步: 提取每个 block 的 W[0]-W[15] ─────
    all_W = []
    # 第一个 block 使用 SHA1 初始 state
    state = tuple(SHA1_H0)

    for blk_idx, blk in enumerate(blocks):
        x14_values = [entry[1] for entry in blk]  # 取 x14
        if len(x14_values) < 16:
            print(f"[!] Block {blk_idx} 只有 {len(x14_values)} 轮 K[0], 需要 >= 16")
            continue

        W = extract_W_from_block(x14_values[:20], state)
        all_W.append(W)

        print(f"\n[*] Block {blk_idx} 消息字 W[0]-W[15]:")
        for j, w in enumerate(W):
            be = w.to_bytes(4, "big")
            asc = "".join(chr(b) if 32 <= b < 127 else "." for b in be)
            print(f"    W[{j:2d}] = 0x{w:08x}  bytes={be.hex()}  |{asc}|")

        # 计算此 block 处理后的 state (用于下一个 block)
        # 用标准 SHA1 对提取的 W 值做一次 block transform
        msg_words = W[:16]
        # 展开 W[16]-W[79]
        Wfull = list(msg_words)
        for t in range(16, 80):
            val = Wfull[t-3] ^ Wfull[t-8] ^ Wfull[t-14] ^ Wfull[t-16]
            Wfull.append(rotl32(val, 1))

        a, b, c, d, e = state
        for t in range(80):
            temp = (rotl32(a, 5) + sha1_f(t, b, c, d) + e +
                    Wfull[t] + sha1_K(t)) & MASK32
            a, b, c, d, e = temp, a, rotl32(b, 30), c, d

        new_state = tuple((s + v) & MASK32
                          for s, v in zip(state, (a, b, c, d, e)))
        print(f"    Block {blk_idx} 后 state: "
              + " ".join(f"{s:08x}" for s in new_state))
        state = new_state

    # ───── 第四步: 解析 padding, 确定消息长度 ─────
    if len(all_W) >= 1:
        last_W = all_W[-1]
        # SHA1 padding: 最后 2 个 word 是 64 位消息长度 (big-endian bits)
        msg_len_bits = (last_W[14] << 32) | last_W[15]
        msg_len_bytes = msg_len_bits // 8
        print(f"\n[*] SHA1 padding 分析:")
        print(f"    W[14] = 0x{last_W[14]:08x}, W[15] = 0x{last_W[15]:08x}")
        print(f"    消息长度 = {msg_len_bits} bits = {msg_len_bytes} bytes")

        # 找到 0x80 padding 标记
        pad_word_idx = None
        for j in range(13, -1, -1):
            w_bytes = last_W[j].to_bytes(4, "big")
            if 0x80 in w_bytes:
                pad_pos = w_bytes.index(0x80)
                # 确认后续字节全为 0
                if all(b == 0 for b in w_bytes[pad_pos + 1:]):
                    pad_word_idx = j
                    break

        if pad_word_idx is not None:
            data_bytes_in_last_block = pad_word_idx * 4
            # 加上 padding word 中 0x80 之前的字节
            w_bytes = last_W[pad_word_idx].to_bytes(4, "big")
            data_bytes_in_last_block += w_bytes.index(0x80)
            total_data_blocks = len(all_W)
            total_from_padding = (total_data_blocks - 1) * 64 + data_bytes_in_last_block
            print(f"    Padding 0x80 位于最后 block 的 W[{pad_word_idx}]")
            print(f"    最后 block 中数据字节数: {data_bytes_in_last_block}")
            print(f"    从 padding 推算总长度: {total_from_padding} bytes")

    # ───── 第五步: 重组完整的 SHA1 输入 ─────
    print(f"\n{'='*60}")
    print(f"SHA1 输入 ({msg_len_bytes} bytes):")
    print(f"{'='*60}")

    full_input = b""
    for blk_idx, W in enumerate(all_W):
        if blk_idx < len(all_W) - 1:
            # 非最后 block: 全部 16 个 word 都是数据
            for w in W:
                full_input += w.to_bytes(4, "big")
        else:
            # 最后一个 block: 只取 padding 之前的数据
            for j in range(16):
                w_bytes = W[j].to_bytes(4, "big")
                if j < pad_word_idx:
                    full_input += w_bytes
                elif j == pad_word_idx:
                    idx = w_bytes.index(0x80)
                    full_input += w_bytes[:idx]
                # j > pad_word_idx: 跳过 (padding / length)

    # 截断到实际消息长度
    full_input = full_input[:msg_len_bytes]

    # Hex dump
    for i in range(0, len(full_input), 16):
        chunk = full_input[i:i + 16]
        hex_part = " ".join(f"{b:02x}" for b in chunk)
        ascii_part = "".join(chr(b) if 32 <= b < 127 else "." for b in chunk)
        print(f"  {i:04x}: {hex_part:<48s} |{ascii_part}|")

    # ───── 第六步: 计算并验证 SHA1 输出 ─────
    computed_hash = hashlib.sha1(full_input).hexdigest()
    final_hash = "".join(f"{s:08x}" for s in state)

    print(f"\n{'='*60}")
    print(f"SHA1 输出:")
    print(f"{'='*60}")

static uint8_t mod255_byte(uint64_t acc) {
    return (uint8_t)((acc + acc / 0xFFULL) & 0xFFULL);
}

static void lcg_hi32_stream(uint64_t seed, uint32_t *out, int n) {
    uint64_t x = seed;
    for (int i = 0; i < n; i++) {
        x = x * LCG_A + LCG_C;
        out[i] = (uint32_t)(x >> 32);
    }
}

static void build_random(
    uint64_t ts_seed,
    uint64_t mask,
    uint8_t *start, uint8_t *end1, uint8_t *end2,
    uint64_t *acc1, uint64_t *acc2
) {
    uint32_t hi[9];
    lcg_hi32_stream(ts_seed, hi, 9);

    uint32_t w10 = (uint32_t)(hi[4] + hi[1]);
    uint32_t w14 = (uint32_t)(hi[5] + hi[2]);
